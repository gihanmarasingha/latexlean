\begin{leancode}
example : ∃ b : ℝ, ∀ x : ℝ, b ≤ x ^ 2 -:e  2 * x := by
  use -1
  intro x
  calc
    -1 ≤ -1 + (x - 1) ^ 2 := by extra
    _ = x ^ 2 - 2 * x := by ring
